Desafio 1
let respostaIncorreta = 0
let respostaCorreta = 0

// Pergunta 1
console.log("Quantos meses há em um ano?")
console.log("A - 12 meses")
console.log("B - 10 meses")
let pergunta1 = prompt("Digite a alternativa correta: ")
switch (true){
    case pergunta1 == "A" || pergunta1 == "12":
        console.log("Alternativa correta!")
        respostaCorreta++
        break;
    default:
        console.log("Alternativa Errada!")
        respostaIncorreta++
        break;
}
